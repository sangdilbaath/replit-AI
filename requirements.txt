def hash_password(password: str) -> str:
    salt = secrets.token_hex(16)
    key = hashlib.pbkdf2_hmac('sha256', password.encode(), salt.encode(), 100000)
    return f"{salt}:{key.hex()}"

def verify_password(stored_hash: str, provided_password: str) -> bool:
    if not stored_hash or ":" not in stored_hash:
        return False
    try:
        salt, stored_key_hex = stored_hash.split(":")
        key = hashlib.pbkdf2_hmac('sha256', provided_password.encode(), salt.encode(), 100000)
        return key.hex() == stored_key_hex
    except Exception:
        return False

@st.cache_resource
def init_connection() -> Optional[Client]:
    try:
        url = st.secrets.get("SUPABASE_URL")
        key = st.secrets.get("SUPABASE_KEY")
        if not url or not key:
            return None
        return create_client(url, key)
    except Exception as e:
        print(f"Supabase error: {e}")
        return None

try:
    supabase = init_connection()
except Exception:
    supabase = None

def _format_user(user: dict) -> dict:
    if user:
        val = user.get("has_payment_on_file", 0)
        if isinstance(val, bool):
            user["has_payment_on_file"] = val
        else:
            user["has_payment_on_file"] = bool(int(val) if str(val).isdigit() else val)
    return user

def get_user(email: str) -> Optional[dict]:
    if not supabase:
        return None
    try:
        response = supabase.table("users").select("*").eq("email", email).execute()
        if response.data:
            return _format_user(response.data[0])
        return None
    except Exception as e:
        print(f"get_user error: {e}")
        return None

def verify_or_create_user(email: str, password: str):
    if not supabase:
        return False
    user = get_user(email)
    if user:
        stored_pw = user.get("password")
        if stored_pw is None:
            try:
                hashed_pw = hash_password(password)
                supabase.table("users").update({"password": hashed_pw}).eq("email", email).execute()
                user["password"] = hashed_pw
                return user
            except Exception:
                return False
        if ":" in stored_pw:
            return user if verify_password(stored_pw, password) else False
        else:
            if stored_pw == password:
                try:
                    hashed_pw = hash_password(password)
                    supabase.table("users").update({"password": hashed_pw}).eq("email", email).execute()
                    user["password"] = hashed_pw
                except Exception:
                    pass
                return user
            return False
    else:
        try:
            hashed_pw = hash_password(password)
            new_user = {
                "email": email,
                "password": hashed_pw,
                "plan_type": "none",
                "has_payment_on_file": False
            }
            res = supabase.table("users").insert(new_user).execute()
            return _format_user(res.data[0]) if res.data else False
        except Exception as e:
            print(f"create_user error: {e}")
            return False

def activate_plan(email: str, plan_type: str) -> Optional[dict]:
    if not supabase:
        return None
    try:
        now = datetime.now()
        if plan_type in ("free_trial", "trial"):
            days = 7 if plan_type == "free_trial" else 2
            trial_end = (now + timedelta(days=days)).strftime("%Y-%m-%d %H:%M:%S")
            supabase.table("users").update({
                "plan_type": plan_type,
                "has_payment_on_file": True,
                "trial_start_date": now.strftime("%Y-%m-%d %H:%M:%S"),
                "trial_end_date": trial_end,
                "expiry_date": trial_end
            }).eq("email", email).execute()
        else:
            supabase.table("users").update({
                "plan_type": plan_type,
                "has_payment_on_file": True,
                "expiry_date": None
            }).eq("email", email).execute()
        return get_user(email)
    except Exception as e:
        print(f"activate_plan error: {e}")
        return None

def admin_create_user(email: str, password: str, plan_type: str, duration_days: int) -> bool:
    if not supabase:
        return False
    try:
        if plan_type.lower() == "trial":
            duration_days = 2
        expiry_date = (datetime.now() + timedelta(days=duration_days)).strftime("%Y-%m-%d %H:%M:%S")
        hashed_pw = hash_password(password)
        user = get_user(email)
        payload = {
            "password": hashed_pw,
            "plan_type": plan_type,
            "has_payment_on_file": True,
            "expiry_date": expiry_date,
            "trial_end_date": None
        }
        if user:
            supabase.table("users").update(payload).eq("email", email).execute()
        else:
            payload["email"] = email
            supabase.table("users").insert(payload).execute()
        return True
    except Exception as e:
        print(f"admin_create_user error: {e}")
        return False

def has_used_trial(email: str) -> bool:
    user = get_user(email)
    if not user:
        return False
    return user.get("trial_start_date") is not None

def is_account_expired(user: dict) -> bool:
    if not user or not user.get("expiry_date"):
        return False
    try:
        end = datetime.strptime(str(user["expiry_date"])[:19], "%Y-%m-%d %H:%M:%S")
        return datetime.now() > end
    except Exception:
        return False

def days_remaining(user: dict) -> int:
    if not user.get("expiry_date"):
        return 0
    try:
        end = datetime.strptime(str(user["expiry_date"])[:19], "%Y-%m-%d %H:%M:%S")
        diff = end - datetime.now()
        return max(0, diff.days)
    except Exception:
        return 0

def get_admin_stats() -> dict:
    if not supabase:
        return {"total_users": 0, "plans": {}}
    try:
        count_res = supabase.table("users").select("id", count="exact").execute()
        total = count_res.count if count_res.count is not None else 0
        plan_res = supabase.table("users").select("plan_type").execute()
        plans = {}
        for row in plan_res.data:
            pt = row.get("plan_type", "none")
            plans[pt] = plans.get(pt, 0) + 1
        return {"total_users": total, "plans": plans}
    except Exception as e:
        print(f"get_admin_stats error: {e}")
        return {"total_users": 0, "plans": {}}

def block_user_trial(email: str) -> bool:
    if not supabase:
        return False
    try:
        yesterday = (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d %H:%M:%S")
        supabase.table("users").update({
            "expiry_date": yesterday,
            "trial_end_date": yesterday
        }).eq("email", email).execute()
        return True
    except Exception as e:
        print(f"block_user_trial error: {e}")
        return False

# ── Analysis History ───────────────────────────────────────────

def save_analysis(email: str, query: str, result_text: str, filename: str = "") -> bool:
    """Save an AI analysis result to the analyses table."""
    if not supabase:
        return False
    try:
        supabase.table("analyses").insert({
            "email": email,
            "query": query[:500],
            "result": result_text[:2000],
            "filename": filename[:200],
            "created_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }).execute()
        return True
    except Exception as e:
        print(f"save_analysis error: {e}")
        return False

def get_user_analyses(email: str, limit: int = 20) -> list:
    """Fetch the most recent analyses for a user."""
    if not supabase:
        return []
    try:
        res = supabase.table("analyses").select("*").eq("email", email)\
            .order("created_at", desc=True).limit(limit).execute()
        return res.data if res.data else []
    except Exception as e:
        print(f"get_user_analyses error: {e}")
        return []

# ── Bookmarks ─────────────────────────────────────────────────

def save_bookmark(email: str, query: str) -> bool:
    """Save a bookmarked query."""
    if not supabase:
        return False
    try:
        existing = supabase.table("bookmarks").select("id")\
            .eq("email", email).eq("query", query[:500]).execute()
        if existing.data:
            return True
        supabase.table("bookmarks").insert({
            "email": email,
            "query": query[:500],
            "created_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }).execute()
        return True
    except Exception as e:
        print(f"save_bookmark error: {e}")
        return False

def get_bookmarks(email: str) -> list:
    """Fetch all bookmarks for a user."""
    if not supabase:
        return []
    try:
        res = supabase.table("bookmarks").select("*")\
            .eq("email", email).order("created_at", desc=True).execute()
        return res.data if res.data else []
    except Exception as e:
        print(f"get_bookmarks error: {e}")
        return []

def delete_bookmark(bookmark_id: int) -> bool:
    """Delete a bookmark by ID."""
    if not supabase:
        return False
    try:
        supabase.table("bookmarks").delete().eq("id", bookmark_id).execute()
        return True
    except Exception as e:
        print(f"delete_bookmark error: {e}")
        return False

def get_all_users(limit: int = 200) -> list:
    """Admin: fetch all user records."""
    if not supabase:
        return []
    try:
        res = supabase.table("users").select(
            "email, plan_type, has_payment_on_file, expiry_date, trial_start_date"
        ).order("email").limit(limit).execute()
        return res.data if res.data else []
    except Exception as e:
        print(f"get_all_users error: {e}")
        return []
